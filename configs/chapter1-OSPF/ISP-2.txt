! ============================================
! Chapter 1. 공중망 구성 (OSPF)
! Device : R4 (ISP-2)
! ============================================

en
conf t
!
no ip domain-lookup
!
hostname ISP-2
!
line con 0
 logging syn
 exec-timeout 0 0
 login
!
line vty 0 4
 password cisco
 login
!
interface loopback 0
 ip address 100.100.12.2 255.255.255.0
 ip ospf network point-to-point
!
interface loopback 120
 ip address 120.20.2.2 255.255.255.0
 ip ospf network point-to-point
!
interface serial 1/0
 no shutdown
 encapsulation frame-relay
 no frame-relay inverse-arp
!
interface serial 1/0.12 point-to-point
 ip address 121.160.12.2 255.255.255.0
 frame-relay interface-dlci 403
!
interface serial 1/0.23 point-to-point
 ip address 121.160.23.2 255.255.255.0
 frame-relay interface-dlci 405
!
router ospf 1
 router-id 22.22.22.22
 passive-interface default
 no passive-interface serial 1/0.12
 no passive-interface serial 1/0.23
 network 100.100.12.0  0.0.0.255 area 0
 network 121.160.12.0  0.0.0.255 area 0
 network 121.160.23.0  0.0.0.255 area 0
 network 120.20.2.0    0.0.0.255 area 0
!
end
write memory
