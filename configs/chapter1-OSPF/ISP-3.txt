! ============================================
! Chapter 1. 공중망 구성 (OSPF)
! Device : R5 (ISP-3)
! ============================================

en
conf t
!
no ip domain-lookup
!
hostname ISP-3
!
line con 0
 logging syn
 exec-timeout 0 0
 login
!
line vty 0 4
 password cisco
 login
!
interface loopback 0
 ip address 100.100.13.3 255.255.255.0
 ip ospf network point-to-point
!
interface loopback 130
 ip address 130.30.3.3 255.255.255.0
 ip ospf network point-to-point
!
interface serial 1/0
 no shutdown
 encapsulation frame-relay
 no frame-relay inverse-arp
!
interface serial 1/0.23 point-to-point
 ip address 121.160.23.3 255.255.255.0
 frame-relay interface-dlci 504
!
interface serial 1/0.34 point-to-point
 ip address 121.160.34.3 255.255.255.0
 frame-relay interface-dlci 506
!
router ospf 1
 router-id 33.33.33.33
 passive-interface default
 no passive-interface serial 1/0.23
 no passive-interface serial 1/0.34
 network 100.100.13.0  0.0.0.255 area 0
 network 121.160.23.0  0.0.0.255 area 0
 network 121.160.34.0  0.0.0.255 area 0
 network 130.30.3.0    0.0.0.255 area 0
!
end
write memory
