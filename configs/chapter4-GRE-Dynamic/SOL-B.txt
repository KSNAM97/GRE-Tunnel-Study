! ============================================
! Chapter 4. GRE Tunnel (Dynamic) + EIGRP
! Device : R2 (SOL-B)
! Tunnel Source/Dest : Loopback 0 IP 사용
! Tunnel Network    : 172.16.100.0/24
! EIGRP AS 100
! ============================================

en
conf t
!
interface tunnel 12
 ip address 172.16.100.2 255.255.255.0
 tunnel source 100.100.2.2
 tunnel destination 100.100.1.1
!
! 사설망 동적 라우팅 (EIGRP)
router eigrp 100
 no auto-summary
 passive-interface default
 no passive-interface tunnel 12
 network 192.168.20.0
 network 100.100.20.0 0.0.0.255
 network 172.16.100.0 0.0.0.255
!
end
write memory
